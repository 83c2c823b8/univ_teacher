	\begin{numcases}
		{}
		x - 3y = -5 \label{first}\\
		2x -4y = -4 \label{second}
	\end{numcases}
\begin{enumerate}
	\item[Step1]\hfill\\
		(\ref{first})$\times$ 2\\
		\begin{equation}
			2x - 6y = -10
		\end{equation}
	\item[Step2]\hfill\\
		(3) $ - $ (2)
			\begin{equation*}
				-2y = -6
			\end{equation*}
	\item[Step3]\hfill\\
		\begin{equation*}
			y = 3
		\end{equation*}
	\item[Step4]\hfill\\
		(1)に$y=3$を代入して，
		\begin{gather*}
			x - 9 = -5\\
			x = 4
		\end{gather*}
	\item[Step5]\hfill\\	
		$(x,y) = (4,3)$を(1),(2)の左辺に代入するとそれぞれ
		\begin{eqnarray*}
			4 - 3\times 3 = -5 \\
			2\times 4 - 4\times 3 = -4
		\end{eqnarray*}
		右辺と一致するからOK
		
\end{enumerate}
